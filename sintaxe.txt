USE WORLD;
DROP DATABASE IF EXISTS BD_DEVFRUIT;
CREATE DATABASE BD_DEVFRUIT;

USE BD_DEVFRUIT;

CREATE TABLE TB_CLIENTE(
ID_CLIENTE     int  NOT NULL  AUTO_INCREMENT,
NOME    	 varchar(50)   		 NOT NULL,
CPF    	 CHAR(11)    		     NOT NULL,
SEXO    	 CHAR(1)   			 NOT NULL,
NASCIMENTO     DATE    			 NOT NULL,
NUMERO    	 CHAR(11)   		 NOT NULL,
EMAIL    	 varchar(100)   	 NOT NULL,
SENHA    	 varchar(100)    	 NOT NULL,
 PRIMARY KEY(ID_CLIENTE)
);

CREATE TABLE TB_ENDERECO(

ID_ENDERECO int NOT NULL AUTO_INCREMENT,
ID_CLIENTE     int    NOT NULL,
RUA_AV    	 varchar (200)    	 NOT NULL,
NUMERO    	 varchar(10)    	 NOT NULL,
COMPLEMENTO varchar(50)   	     	 NULL,
BAIRRO    	 varchar(50)   		 NOT NULL,
CEP    	 CHAR(8)   			     NOT NULL,
CIDADE    	 varchar(100)    	 NOT NULL,
UF    		 CHAR(2)   			 NOT NULL,
PRIMARY KEY (ID_ENDERECO),
FOREIGN KEY (ID_CLIENTE) REFERENCES TB_CLIENTE(ID_CLIENTE)
);
CREATE TABLE TB_TIPO(
ID_TIPO   	 int  NOT NULL AUTO_INCREMENT,
DESCRICAO     varchar(200)		 NOT NULL,
PRIMARY KEY (ID_TIPO)
);

CREATE TABLE TB_PRODUTO(
ID_PRODUTO    int 	 NOT NULL AUTO_INCREMENT ,
ID_TIPO     int   					 NOT NULL,
DESCRIÇÃO     varchar(200)   		 NOT NULL,
VALOR_UNIT     REAL (6,2)   		 NOT NULL,
QTD_ESTOQUE int         			 NOT NULL,
VALOR_COMPRA REAL (6,2)              NOT NULL,
VALOR_VENDA REAL (6,2)               NOT NULL,
DISPONIVEL CHAR(1)                   NOT NULL,
PRIMARY KEY (ID_PRODUTO),
FOREIGN KEY(ID_TIPO) REFERENCES TB_TIPO(ID_TIPO)
);

CREATE TABLE TB_PEDIDO(
ID_PEDIDO 	 int  	 NOT NULL AUTO_INCREMENT,
ID_CLIENTE     int   		     NOT NULL,
ID_PRODUTO    int   	         NOT NULL,
ID_TIPO   	 int   	             NOT NULL,
DATA_PEDIDO DATE        		 NOT NULL,
HORA_PEDIDO TIME    			 NOT NULL,
QTD   		 REAL(6,2)    		 NOT NULL,
VALOR    	 REAL (6,2)    		 NOT NULL,
PRIMARY KEY (ID_PEDIDO),
FOREIGN KEY (ID_CLIENTE) REFERENCES TB_CLIENTE(ID_CLIENTE),
FOREIGN KEY (ID_TIPO)    REFERENCES TB_TIPO(ID_TIPO),
FOREIGN KEY (ID_PRODUTO) REFERENCES TB_PRODUTO(ID_PRODUTO)
);

CREATE TABLE TB_ENTREGA(
ID_ENTREGA   		 int  	     NOT NULL AUTO_INCREMENT,
ID_CLIENTE   		 int   		 NOT NULL,
ID_PEDIDO   		 int   		 NOT NULL,
DATA_HORA_ENTREGA     DATETIME   NOT NULL,
PRIMARY KEY (ID_ENTREGA),
FOREIGN KEY (ID_CLIENTE) REFERENCES TB_CLIENTE(ID_CLIENTE)
);
